DROP TABLE IF EXISTS User;
CREATE Table IF NOT EXISTS User(
	uId INT AUTO_INCREMENT NOT NULL,
	name VARCHAR(200),
	contact VARCHAR(20),
	email VARCHAR(200),
	date DATE,
	time TIME,
	CONSTRAINT PRIMARY KEY(uId)
);

DROP TABLE IF EXISTS Room;
CREATE Table IF NOT EXISTS Room(
	rId INT AUTO_INCREMENT NOT NULL,
	uId INT,
	date DATE,
	time TIME,
	CONSTRAINT PRIMARY KEY(rId),
	CONSTRAINT FOREIGN KEY(uId) REFERENCES User(uId)
        	ON DELETE CASCADE ON UPDATE CASCADE
);

DROP TABLE IF EXISTS Message;
CREATE Table IF NOT EXISTS Message(
	mId INT AUTO_INCREMENT NOT NULL,
	rId INT,
	user VARCHAR(50),
	message TEXT,
	date DATE,
	time TIME,
	CONSTRAINT PRIMARY KEY(mId),
	CONSTRAINT FOREIGN KEY(rId) REFERENCES Room(rId)
        	ON DELETE CASCADE ON UPDATE CASCADE
);
